#:ed.ncar			| lugar del cursor
#:ed.ipan			| inicio de pantalla
#:ed.nombre )( 256
#:ed.nlin			| numero de linea (old)

::ed.load
	'ed.ncar "mem/edit-code.mem" load drop ;

::ed.save
	'ed.ncar 268 "mem/edit-code.mem" save ;

::ed.path | -- str
	mark
	0 'ed.nombre ( c@+ 1? )(
		$2f =? ( rot drop here rot rot ) | /
		,c ) ,c drop
	0? ( drop empty here ; )
	0 swap 1+ c!
	empty here ;
