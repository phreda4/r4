:shader | version 1
	@+ $7fffffff =? ( drop 4 a+ 4+ ; )
	8 >> $f and 1+
	swap @+ $f0f0f0 and
	rot *
	a!+ ;

:shader | version 2
	@+ $7fffffff =? ( drop 4 a+ 4+ ; )
	swap @+ 4 >> $f0f0f and
	rot
	pick2 @ - 5 >> 7 + 15 clamp0max 1+
	* $f0f0f0 and
	dup 4 >> or
	a!+ ;

:shader | version 3
	@+ $7fffffff =? ( drop 4 a+ 4+ ; )
	swap >b
	b@+ 4 >> $f0f0f and
	swap | color z
	dup b@ -
	b> 16 - @ rot - +
	6 >> 7 + 15 clamp0max 1+
	* $f0f0f0 and
	dup 4 >> or
	a!+ b> ;

::zdraw | x y --
	setxy
	sw w3do - 2 <<
	zb
	h3do ( 1? )( 1-
		w3do ( 1? )( 1-
			rot shader rot rot
			) drop
		pick2 a+
		) 3drop ;

:drawz | z --
	sizx neg sizy neg pick2 3dop
	sizx sizy neg pick2 3dline
	sizx sizy pick2 3dline
	sizx neg sizy pick2 3dline
	sizx neg sizy neg rot 3dline ;

:drawl | x1 x2 --
	2dup sizz neg 3dop sizz 3dline ;

:box3d | sx sy sz --
	'sizz ! 'sizy ! 'sizx !
	sizz neg drawz sizz drawz
	sizx neg sizy neg drawl sizx sizy neg drawl
	sizx sizy drawl sizx neg sizy drawl
	;
