#tanques 0 0
#misiles 0 0
#piedras 0 0
#fx 0 0

|-----------------
:r00001 rand 0.0001 mod ;
:r0001 rand 0.001 mod ;
:r001 rand 0.01 mod ;
:r01 rand 0.1 mod ;

:fpos
	8 >> sh 2/ + swap 8 >> sw 2/ + swap pos ;

#tano 0 0 tano1 10 tano2 10 tano3 10 0

:secani | 'secani -- 'spr
	>r r @ 0? ( r 8 + r!+ r!+ r> @ ; )
	r 4+ @ 1+ over 4+ @ | 'spr cnt max
	<? ( r> 4+ ! @ ; ) drop
	8 + dup @ 0? ( 2drop r 8 + dup @ ) | adr spr
	swap r!+ 0 r> ! ;


|-------------------------
:tbox | adr --
	>r
	r@+ 'tx1 ! r@+ 'ty1 !
	r@+ 'tx2 ! r@+ 'ty2 !
	r@+ drop |fillbox
	r@+ dup $ffff and swap 16 >> $ffff and boxpad
	r@+ dup 12 >> $f and swap $fff and setfont
	r@+ r@+ dup ink 24 >> boxprintn
	rdrop ;


:+tbox | "" w h x y  --
	'tbox 'fx p!+ >r
	over pick4 - r!+
	dup pick3 - r!+
	rot + rot rot + r!+ r!+
	0 r!+ | fondo
	0 r!+	| pad
	$1032 r!+ | font
	r!+
	$a00ff00 r!+
	rdrop	;

|-----------------
:phumo
	dup @ 1- 0? ( nip ; )
	dup 2* alpha
	swap !+ >r
	r@+	r@+ fpos
	r@+ r 12 - +!	| VX
	r@+ r 12 - +!	| VY
	r@+ qdim
	r@+ r@+ rnsprite
	r@+ r> 8 - +!
	255 alpha
	;

:+humo | x y --
	'phumo 'fx p!+ >r
	127 r!+ | vida
	r!+ r!+ | x y
	r0001 r!+ r0001 r!+ | vx vy
	r0001 4 + r!+		| tamaño
	'humo2 r!+			| dibujo
	0 r!+
	r001 r!+
	rdrop ;

:+estela | x y --
	'phumo 'fx p!+ >r
	64 r!+
	r!+ r!+ | x y
	r0001 r!+ r0001 r!+
	r00001 10 + r!+		| tamaño
	'humo2 r!+			| dibujo
	0 r!+
	r001 r!+
	rdrop ;

:golpet1
|	0 'tb2 ! 0 'vt !
|	10 'danio +!
|	danio 100 >=? ( ganotanque2 exit ) drop
|	10 ( 1? )( yt xt +humo 1- ) drop
	;

:golpet2
|+	0 'tb ! 0 'vt2 !
|	10 'danio2 +!
|	danio2 100 >=? ( ganotanque1 exit ) drop
|	10 ( 1? )( yt2 xt2 +humo 1- ) drop
	;

:hitmisil | x y tanque -- 0
    "Hit" 2over fpos 50 20 xc yc +tbox
	5 ( 1? )( 1-
		pick3 pick3 swap
		+humo )
	4drop 0 ;

:pmisil
	dup @ 1- 0? ( nip ; )	| tiempo de vida
	swap !+ >r
	r@+ r@+
|	rnd $1000 nand? ( pick2 pick2 swap +estela ) drop
	2dup 'tanques 0.05 p.in2d? 1? ( hitmisil rdrop ; ) drop
	fpos 48 qdim
	'misil r@+ rnsprite
	r@+ r 16 - +!
	r@+ r 16 - +!
	rdrop
	;

:+misil
	dup >r r@+ r@+ r> 4+ @	| x y d
	'pmisil 'misiles p!+ >r
	170 r!+
	rot r!+ swap r!+
	dup r!+
	sincos 0.01 *. r!+ 0.01 *. r!+
	-8 r+
	r@+ 10* r 16 - +!
	r@+ 10* r 16 - +!
	rdrop ;


|-------------------------
:oruga
	over
	1? ( 'tano secani )( 'tano1 )
	nip over rnsprite ;

:drawtan | rot --
	oruga
	'tanc over rnsprite
	'tant over rnsprite
 	;

:drawtan2 | rot --
	oruga
	'tanc2 over rnsprite
	'tant over rnsprite
 	;

:drawtan3 | rot --
	oruga
	'tanc3 over rnsprite
	'tant over rnsprite
 	;


| control de tanque
| rigth
| left
| up
| fire

:tcontrol
|	%1 and? ( -0.001 turn )
|	%10 and? ( 0.001 turn )
|	%100 and? ( 0.001 adv )
|	%1000 and? ( +misil )
	;

| tanque 1
| x y dibujo dt
| (adt) (vt)
| daño energia id
:t1
	>r
	80 qdim
	r@+	r@+ fpos
	r> 8 +
	[ -0.001 over ! ; ] <le>
	[ 0.001 over ! ; ] <ri>
	[ 0 over ! ; ] >le<
	[ 0 over ! ; ] >ri<
	4+
	[ 0.001 over ! ; ] <up>
	[ 0 over ! ; ] >up<
	4 - >r
	r@+ r 8 - +!
	r dup @ swap 8 - @ drawtan | vt dt
	sincos pick2 *. r 20 - +! *. r 16 - +!
    r>
	20 - '+misil <dn>
	drop
	;

:t2
	>r
	80 qdim
	r@+	r@+ fpos
	r> 8 +
	[ -0.001 over ! ; ] 30 ?key
	[ 0.001 over ! ; ] 32 ?key
	[ 0 over ! ; ] 30 ?ukey
	[ 0 over ! ; ] 32 ?ukey
	4+
	[ 0.001 over ! ; ] 17 ?key
	[ 0 over ! ; ] 17 ?ukey
	4 - >r
	r@+ r 8 - +!
	r dup @ swap 8 - @ drawtan2 | vt dt
	sincos pick2 *. r 20 - +! *. r 16 - +!
	r> 20 -
	'+misil 31 ?key
	drop
	;

| tanque de prueba robot
:t3
	>r
	80 qdim
	r@+	r@+ fpos
	r> 8 +
	[ -0.001 over ! ; ] 30 ?key
	[ 0.001 over ! ; ] 32 ?key
	[ 0 over ! ; ] 30 ?ukey
	[ 0 over ! ; ] 32 ?ukey
	4+
	[ 0.001 over ! ; ] 17 ?key
	[ 0 over ! ; ] 17 ?ukey
	4 - >r
	r@+ r 8 - +!
	r dup @ swap 8 - @ drawtan3 | vt dt
	sincos pick2 *. r 20 - +! *. r 16 - +!
	r> 20 -
	'+misil 31 ?key
	drop
	;


:reset
	'tanques p.clear
	'piedras p.clear
	'fx p.clear

	't1 'tanques p!+ >r
	0 0 0 0 'stanque 0 -0.5
	r!+ r!+ r!+ r!+ r!+ r!+
	0 r!+
	100 r!+	| energia
	1 r!+ | id tanque
	rdrop

	't2 'tanques p!+ >r
	0 0 0 0.5 'stanque 0 0.5
	r!+ r!+ r!+ r!+ r!+ r!+
	0 r!+
	100 r!+	| energia
	2 r!+ | id tanque
	rdrop

	't3 'tanques p!+ >r
	0 0 0 0.25 'stanque 0.3 0.5
	r!+ r!+ r!+ r!+ r!+ r!+
	0 r!+
	100 r!+	| energia
	3 r!+ | id tanque
	rdrop

	;

	mark
	10 'tanques p.ini
	100 'misiles p.ini
	100 'piedras p.ini
	1000 'fx p.ini
	;

: memoria main ;
