:editando
	usogui
	0 0 fpos
	zoom 1.3 + sh 16 *>> dup 'w ! 'h !
	savebox
	show clrscr
		32 gc.top $ccc gc.hfill
		font-vard-12-bold home cr2
		verde dup " :R%d" print
		blanco actual " EDIT TEX %d" print


		26 gc.botton $ccc gc.hfill
		font-vard-8-bold home cr2

|		0.8 0.8 fdim 0 0 fpos vistamode

		actual drawfix
 
	'exit >esc<
