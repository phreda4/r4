:printpalo | nro --
	0 =? ( "Oro" print drop ; )
	1 =? ( "Copa" print drop ; )
	2 =? ( "Basto" print drop ; )
	3 =? ( "Espada" print drop ; )
	drop
	;

:printcarta | nro --
	4 * dup 'cartas + @ "%d de " print
	'palos + @ printpalo
	;

:printmazo
	0 ( 40 <? )( 
		dup printcarta cr
		1+ ) drop
 	;

:intercambia | a b --
	over 4 * 'cartas + @ | a b d
	over 4 * 'cartas + @  | a b na nb
	pick3 4 * 'cartas + !
	over 4 * 'cartas + !
	over 4 * 'palos + @
	over 4 * 'palos + @ | a b pa pb
	pick3 4 * 'palos + !
	over 4 * 'palos + !
	2drop
	;

:mezclar
	0 ( 100 <? )(
		rand 40 mod abs 
		rand 40 mod abs 
		intercambia
		1+ ) drop 
	;

		;

:drawcarta | palo nro -- 
	mpush
	'carta 3dnsprite
	1- 2 << 'tipos + @
	@+ dup 0 mscalei
	( @+ -1 <>? )(
		mpush
		over @ 0 mtransi
		over 3dnsprite
		mpop
		4+ ) 3drop
	mpop ;

#dibujos 'oro 'copa 'basto 'espada

:drawcartam | nro --
	4 * dup
	'palos + @ 
	2 << 'dibujos + @
	swap 'cartas + @ 
	drawcarta 
	;

:printmazog
	-5.0 -3.0 0 mtransi
	0
	4 ( 1? )( 1-
		10 ( 1? )( 1-
			1.0 0 0 mtransi
			rot dup drawcartam 1+ rot rot
			) drop
		-10.0 2.0 0 mtransi
		) drop
	;


:freelook
	xymouse
	sh 2/ - 7 << swap
	sw 2/ - neg 7 << swap
	neg mrotx
	mroty ;
