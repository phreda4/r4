#tree
#tree>

:loadtree | "" --
	here dup 'tree !
	swap load
	dup 'tree> !
	0 swap !+ 'here ! ;

:+dom | "dom" --
	"%s~" mprint
	tree>
	( 1- dup c@ 0? )( drop 1- ) drop 1+
	|( 1- dup c@ $7e =? )( drop 1- ) drop 1+
	strcpyl 1- 'tree> !
:savetree | "" --
	tree count swap save
	;

:+list | "" lugar arbol --
	mark
	( over <? )(
		c@+ ,c )
	rot ,s
	"|" ,s
	,s
	
	empty
	;
