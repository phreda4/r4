#spr1
	'coso1 0 0.0 0.0 0.3 0.0
	'coso2 0.2 0.0 0.0 0.2 0.0
	0

:draw1part | dir spr -- dir+
	swap >r
	| calculo aqui
	r @ 1+
	dup 7 << sincos 2 >> r 4+ ! 2 >> r 8 + !	| movimiento en circulo
	0.01 r 16 + +!
	r!+
	r@+ r@+ fpos
	r@+ dup fdim
	r@+ rnsprite
	r> ;

:drawpart | tabla --
	( @+ 1? )( draw1part ) 2drop ;
