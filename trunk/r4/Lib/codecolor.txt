:lineacom | adr c -- adr++
	( 13 =? ( drop 1- ; )
	  emit c@+ 0? )
	drop 1- ;

:palstr | adr c -- adr++
	( emit c@+
	  34 =? ( emit ; )
	  0? )
	drop 1- ;

:npal
	( $ff and 32 >? )( emit c@+ )
	drop 1- ;

:palabracol | adr c -- ar..
	32 =? ( emit ; )
	9 =? ( emit ; )
	10 =? ( drop ; )
	$5e =? ( amarillo npal ; )	| $5e ^  Include
	$7c =? ( gris lineacom ; )	| $7c |	 Comentario
	$3A =? ( rojo npal ; )		| $3a :  Definicion
	$23 =? ( violeta npal ; )	| $23 #  Variable
	$22 =? ( blanco palstr ; )	| $22 "	 Cadena
	$27 =? ( cyan npal ; )		| $27 ' Direccion
	over 1- isNro 1? ( drop amarillo npal ; ) drop
|	over ?macro 1? ( 2drop naranja ) drop		| macro
	verde npal ;

::>>lineacolor>> | adr -- adr++
	( c@+ 1? )(
		13 =? ( drop ; )
		palabracol
		) drop 1- ;

::>>lineacolor0 | adr -- adr++/0
	( c@+ 1? )(
		13 =? ( drop ; )
		palabracol
		) nip ;

::code.lines | adr -- adr..
	rows swap 0
	( pick2 <? )( swap
		>>lineacolor>> cr
		swap 1+ )
	drop nip ;
