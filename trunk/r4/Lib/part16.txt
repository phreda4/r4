::p.ini | cantidad list --
	here dup rot !+ ! 6 << 'here +! ;

::p.clear | list --
	dup 4+ @ swap ! ;

::p!+ | 'act list -- adr
	dup >r @ !+
	64 r> +! ;

:delp | list end now -- list end- now-
	swap 64 - 2dup 16 move
	dup pick3 !
	swap 64 - ;

::p.draw | list --
	dup @+ swap @
	( over <? )(
		dup @+ exec 0? ( drop delp )
		64 + ) 3drop ;

::p.del | adr list --
	>r r @ 64 - 16 move r> dup @ 64 - swap ! ;

::p.mapv | 'vector list --
	@+ swap @
	( over <? )(
		pick2 exec
		64 + ) 3drop ;
