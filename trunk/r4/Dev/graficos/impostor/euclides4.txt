|----------- graficacion
#rx
#ry
:freelook
	xymouse
	sh 2/ - 7 << swap
	sw 2/ - neg 7 << swap
	neg 'rx ! 'ry ! ;


:drawp | x y z ---
	project3d |2dup op line ;
	-? ( 2drop ; ) sh >=? ( 2drop ; )
	swap -? ( 2drop ; ) sw >=? ( 2drop ; ) swap
	setxy
	ink@ px!+
	;

#nodoa 0

:drawpc | '3do --
	@+ >r @
	( 1? )( 1-
		r nodoa =? ( blanco )( azul ) drop
		r@+ invmorton3d drawp
		4 r+ ) drop
	rdrop ;


:drawp2
	project3d
	2dup op
	over 1+ over line
	over 1+ over 1+ line
	over over 1+ line
	line ;

:drawpcf | '3do --
	@+ >r @
	( 1? )( 1-
		r@+ r@+ ink
		invmorton3d drawp2
		) drop
	rdrop ;

:3dop project3d op ;
:3dline project3d line ;

|-------- dibujo octree
#xc #yc #zc
#size

:drawz | z --
	xc size - yc size - pick2 3dop
	xc size + yc size - pick2 3dline
	xc size + yc size + pick2 3dline
	xc size - yc size + pick2 3dline
	xc size - yc size - rot 3dline ;

:drawl | x1 x2 --
	2dup zc size - 3dop zc size + 3dline ;

:box3d
	zc size - drawz zc size + drawz
	xc size - yc size - drawl xc size + yc size - drawl
	xc size + yc size + drawl xc size - yc size + drawl ;

|-------------- muestra oc de 1 punto
:+movx
	1 and? ( size )( size neg ) 'xc +!
	2 and? ( size )( size neg ) 'yc +!
	4 and? ( size )( size neg ) 'zc +!
	drop ;

:drawtree | nro shift --
	size 2/ 'size !
	2dup >> $7 and | nro shift val
	+movx box3d
	3 - 0? ( 2drop ; )
	drawtree ;

:showpoint | adr --
	0? ( drop ; ) @
	0 'xc ! 0 'yc ! 0 'zc !
	$1ff 'size ! box3d
	27 drawtree ;

:duchaso | adr --
	@+ >r @ dup 'cnt !
	here 'listax !
	dup 3 << 'here +!
	here 'listay !
	dup 3 << 'here +!
	here 'listaz !
	dup 3 << 'here +!
	( 1? )( 1-
		r@+ r@+ drop
		invmorton3d | x y z
		pick3 3 << listaz + !
		pick2 3 << listay + !
		over 3 << listax + !
		) drop
	rdrop
	cnt 1+ listax shellsort
	cnt 1+ listay shellsort
	cnt 1+ listaz shellsort
	listax >r 0 'cntx !
	cnt ( 1? )( 1-
		r@+ r @ <>? ( 1 'cntx +! )
		drop ) drop rdrop
	listay >r 0 'cnty !
	cnt ( 1? )( 1-
		r@+ r @ <>? ( 1 'cnty +! )
		drop ) drop rdrop
	listaz >r 0 'cntz !
	cnt ( 1? )( 1-
		r@+ r @ <>? ( 1 'cntz +! )
		drop ) drop rdrop
	| y??

	;
