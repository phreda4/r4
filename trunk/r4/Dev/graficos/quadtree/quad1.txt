:drawimagen
	imagen >r
	res ( 1? )( 1-
		res ( 1? )( 1-
			r@+ px!+
			) drop
		sw res - px+!
		) drop
	rdrop
	;

:copyimg
	imagen >r
	res ( 1? )( 1-
		res ( 1? )( 1-
			px@ r!+ 1 px+!
			) drop
		sw res - px+!
		) drop
	rdrop
	;

:+pnt | x y --
	morton2d puntos> swap puntos> !+ !+ 'puntos> ! ;

:makecuadtree
|--- agrego puntos
	imagen >r
	0 ( res <? )(
		0 ( res <? )(
			r@+ 1? ( drop 2dup swap +pnt )( drop )
			1+ ) drop
		1+ ) drop
	rdrop
|--- ordeno
	puntos> puntos - 3 >> 1+ puntos 2dup shellsort
|---- nivel0
	'listap !+ !
	;

#xx #yy #size

:box2d
	xx size - yy size - 2dup op
	over yy size + line
	xx size + yy size + line
	xx size + over line
	line ;

:+mov | id --
	1 and? ( size )( size neg ) 'xx +!
	2 and? ( size )( size neg ) 'yy +!
	drop ;

:drawtree | nro shift --
	size 2/ 'size !
	2dup >> $3 and | nro shift val
	+mov
	box2d
	0? ( 2drop ; )
	2 - drawtree ;

:drawanumber | nro --
	128 dup 'xx ! 'yy !
	res 2/ 'size !
	14 drawtree ; | 16 bits -2 =14 para correr la primera vez

:searchm | valor -- punto
	puntos
	dup 'cer !
	dup @ pick2 - abs 'vcer !
	( puntos> <? )( @+
		pick2 - abs vcer <? ( 'vcer ! dup 4 - 'cer ! )( drop )
		4+ ) 2drop
	cer
	;

:searchm2 | valor -- punto
	puntos
	dup 'cer !
	dup @ pick2 xor 'vcer !
	( puntos> <? )( @+
		pick2 xor vcer <? ( 'vcer ! dup 4 - 'cer ! )( drop )
		4+ ) 2drop
	cer
	;

:distcomp | z z -- d
	invmorton2d rot invmorton2d | x y x y
	rot - dup * >r
	- dup * r> + ;

:searchbf | valor -- punto
	puntos
	dup 'cer !
	dup @ pick2 distcomp 'vcer !
	( puntos> <? )( @+
		pick2 distcomp vcer <? ( 'vcer ! dup 4 - 'cer ! )( drop )
		4+ ) 2drop
	cer
	;

:testalgo | valor -- adr
	dup 'listap bsearchqe 0? ( trace ) dup 'cer !
	@ over distcomp 'vcer !

	dup cer xor
	'listap bsearchqe | valor cn ns
	1? ( dup @ pick3 distcomp
			vcer <? ( 'vcer ! 'cer ! )( 2drop )
			)( drop )
	cer
	;
