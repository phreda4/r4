#numPoints 32

#nsimbol 0
#simbols )( 1024 | 256 simbolos
#infosim )( 1024 | 256 info sim
#lensim )( 1024 | 256 info sim

#memsim
#memsim>

#pc2d )( $3fff
#pc2d> 'pc2d
#id 1
#aux )( $3fff
#aux> 'aux
| formato de trazo
|  X    Y   ID
| $fff fff ff

:xy>n | x y -- n
	$fff and 8 << swap $fff and 20 <<  or id or ;
:n>xy	 | n -- x y
	dup 20 >> swap 12 << 20 >> ; | con signo

:ngc>xy	 | n -- x y ;gc
	dup 20 >> w 12 *>> xc + swap 12 << 20 >> h 12 *>> yc + ; | con signo

:++pc2d | px --
	pc2d> 4 - @ =? ( drop ; )
:+pc2d | px
	pc2d> !+ 'pc2d> ! ;

:pc2dreset
	'pc2d 'pc2d> ! 1 'id ! ;

:pc2ddraw
	0 >r
	'pc2d ( pc2d> <? )( @+
		dup $ff and r <>? ( drop n>xy op 1 r+ )( drop n>xy line )
		) drop rdrop ;

:pc2dpoint
	'pc2d ( pc2d> <? )( @+
		n>xy 2 box
		) drop ;


:pc2dcopy | adr -- adr'
	'pc2d ( pc2d> <? )(
		@+ rot !+ swap ) drop
	0 swap !+ ;

:pntdist | p1 p2 -- len
	n>xy rot n>xy	| x1 y1 x2 y2
	rot - dup * >r
	- dup * r> + sqrt ;

:pc2dlen | -- length
	0 'pc2d
	( pc2d> <? )( @+
		over 4+ @	| adr' p1 p2
		over $ff and over $ff and - | adr' p1 p2 id
		0? ( drop pntdist rot + swap )( 3drop )
		) drop
	sqrt ;

:pc2dlen2 | -- length
	0 'pc2d
	( pc2d> <? )( @+
		over 4+ @	| adr' p1 p2
		over $ff and over $ff and - | adr' p1 p2 id
		0? ( drop pntdist rot + swap )( 3drop )
		) drop ;

#cx #cy #minx #maxx #miny #maxy
#size

:normalize
	'pc2d
	@+ n>xy
	dup 'miny ! 'maxy !
	dup 'minx ! 'maxx !
	( pc2d> <? )( @+ n>xy
		miny <? ( dup 'miny ! )
		maxy >? ( dup 'maxy ! )
		drop
		minx <? ( dup 'minx ! )
		maxx >? ( dup 'maxx ! )
		drop
		) drop
	maxx minx - maxy miny - max 'size !
	minx maxx + 2/ neg 'cx !
	miny maxy + 2/ neg 'cy !
	'pc2d ( pc2d> <? )( dup @
		dup $ff and >r
		n>xy
		cy + $7ff size */
		swap
		cx + $7ff size */
		swap
		xy>n $ffffff00 and r> or
		swap !+ ) drop ;

		allowchome
		) drop
	;


:dumptest

		size "size:%d " print cr
		over "sumaa %d " print cr
		drawresample
		cr
		cyan
		'exit >esc<
		;

|------------ version 1
:calcxy	| p1 p2 n -- nxy
	>r n>xy rot n>xy	| x2 y2 x1 y1
	rot 				| x2 x1 y1 y2
	over - r 16 *>> +	| x2 x1 yn
    $fff and 8 <<
	swap rot
	over - r> 16 *>> +	| yn xn
	$fff and 20 << or ;

:interpola | adr value adr -- adr value
	@+ swap @+ swap @		| value p1 p2 sumalast
	pick3 swap -			| value p1 p2 x1
	size over - 			| value p1 p2 x1 size-x1
	dup >r
	pick4 rot - 16 <</		| value p1 p2 t
	calcxy	| adr value npx

	nip     		| adr npx
	swap 12 -
	r> neg over 8 + +! | disminuye distancia
	2dup ! | npx adr. ; guarda nuevo punto
	swap

	+pc2d
	0 ;

:resample
	'aux >r
	0 'pc2d
	( pc2d> 4 - <? )( @+
		over @	| adr' p1 p2
		over $ff and over $ff and - | adr' p1 p2 id
		0? ( drop 2dup swap r!+ r!+ pntdist dup r!+ rot + swap )( 3drop )
		) drop
	numPoints 1- / 'size !
	r> 'aux> !
	pc2dreset
	'aux @+ +pc2d
	0 swap ( aux> <? )(
		4+ @+ 	| acc adr dist
		rot +	| adr acc+
		size >=? ( over 12 - interpola  )
		swap 4+ ) 2drop

|	pc2d> 'pc2d - 2 >>
|	32 >? ( dup 32 - 2 << neg 'pc2d> +! )

|	( 32 <? )( aux> 8 - @ +pc2d 1+ ) 2drop

	pc2d> 'pc2d - 2 >>
	nsimbol 2 << 'lensim + !

	0 pc2d> !
	;

:gcdraw | adr --
	0 >r
	( @+ 1? )(
		ngc>xy 1 box
|		dup $ff and r <>? ( drop ngc>xy op 1 r+ )( drop ngc>xy line )
		) 2drop rdrop ;

:drawnsimbol | n -- n
	nsimbol >=? ( ; )
	gc.box
	dup 2 << 'simbols + @
	gcdraw
	home
	dup 2 << 'infosim + @ "%d" print cr
	dup 2 << 'lensim + @ "%d" print
	;


:drawsim
	verde
	0.1 %s qdim
	0
	-0.9 ( 0.9 <? )(
		dup -0.8 fpos
		swap drawnsimbol
		1+ swap 0.12 + ) 2drop
	;

#actual

:getxy1 | nro -- xy
	2 << actual 2 << 'simbols + @ + @ ngc>xy ;

:getxy2 | nro -- xy
	2 << 'pc2d + @ ngc>xy ;

:drawmatch
	0 ( 32 <? )(
		dup getxy1 op
		dup getxy2 line
		1+ ) drop
	;

|--------------------------------
:calcfit | nro -- nro suma
	dup 2 << 'simbols + @ >r	| nro   r:simbolo
	'pc2d dup 128 +		| nro act last
	0 rot
	( pick2 <? )( | nro last suma act
		@+ n>xy r@+ n>xy
		rot - rot rot - distfast	| nro last suma act delta
		rot + swap )				| nro last suma act
	drop nip rdrop ;

#best
#nbest

:fitness
	$7fffffff 'best !
	0 ( nsimbol <? )(
		calcfit dup pick2 2 << 'infosim + !
		best <? ( 'best ! dup 'nbest ! )( drop )
		1+ ) drop ;


:searchar
	resample
	normalize
	fitness
	nbest 'actual !
	show clrscr
		dup "%d " print cr
		0.5 %s qdim
		0 dup fpos
		azul gc.box
		drawmatch

		verde actual 2 << 'simbols + @ gcdraw
		cyan 'pc2d gcdraw

		drawsim

		[ pc2dreset exit ; ] >esc< ;

:addchar
	pc2d> 'pc2d - 2 >> 2 <? ( drop pc2dreset ; ) drop
	normalize
	resample
	memsim> dup pc2dcopy 'memsim> !
	nsimbol 2 << 'simbols + !
	1 'nsimbol +!
	pc2dreset
	;

	mark

		'searchar dup <f2> "?search" sp btnt
		'resample dup <f3> "resample" sp btnt

		cr cr
		dup "%d " print
|		pc2dlen dup "len %d " print numPoints 1- / "len med %d " print cr
|		pc2dlen2 dup "len2 %d " print numPoints 1- / "len med %d " print
		cr

		cminiflecha
