|------ animaciones
#secquieto 0 0 p4 10 0
#seccamina 0 0 p1 10 p2 10 p1 10 p3 10 0
#seccaminaf 0 0 p4 10 p5 10 p4 10 p6 10 0
#seccaminad 0 0 p7 10 p8 10 p7 10 p9 10 0


#secver 0 0 ver 20 ver1 20 ver2 5 0
#secusa 0 0 usar 14 usar2 10 0

#pantallas 'pant1 'pant2

#mensaje )( 64
:mensaje! 'mensaje strcpy ;

:secani | 'secani -- 'spr
	>r r @ 0? ( r 8 + r!+ r!+ r> @ ; )
	r 4+ @ 1+ over 4+ @ | 'spr cnt max
	<? ( r> 4+ ! @ ; ) drop
	8 + dup @ 0? ( 2drop r 8 + dup @ ) | adr spr
	swap r!+ 0 r> ! ;

#objetos 0 0
#walls	0 0
#fxs	0 0

|----------- protagonista
#animap 'secquieto
#xp #yp			| posicion
#dd				| direccion
#vx #vy
#yd #xd
#dz 200
#pant 'pant1
|------------ paredes
#xmin -0.8 #xmax 0.8
#ymin -0.8 #ymax 0.8

:quieto!
	'secquieto 'animap ! 0 0 'vy ! 'vx ! ;

:personaje
|-----------------rumbo
	xd xp - yd yp - 2dup atan2 0.004 polar 'vy ! 'vx !
	distfast 0.01 <? ( quieto! ) drop
|-----------------dibujo y animacion
	xp yp fpos
	dz 200 dim
	animap secani nsprite
	vx 'xp +!
	vy 'yp +!
|-----------------paredes
|	xp xmin <? ( xmin 'xp ! quieto! ) xmax >? ( xmax 'xp ! quieto! ) drop
|	yp ymin <? ( ymin 'yp ! quieto! ) ymax >? ( ymax 'yp ! quieto! ) drop
	;

|--------- menu
#menuactivo 0
#menuobj 0

:drawmenu
	menuactivo 0? ( drop ; ) drop
	190 100 dim
	xp yp 0.2 + fpos 'pensar nsprite
	80 dup dim
	xp 0.08 - yp 0.2 + fpos
	[ 'secver secani nsprite ; ]
	[ 'ver nsprite ; ] guiIO
	[ menuobj 24 + @ exec  ; ] guiBtn

	xp 0.08 + yp 0.2 + fpos
	[ 'secusa secani nsprite ; ]
	[ 'usar nsprite ; ] guiIO
	[ menuobj 28 + @ exec ; ] guiBtn

	;

:menuclick
	0 'menuactivo !
	;

|--------- Objetos
:obj
	>r
	r@+ r@+ fpos
	r@+ dup dim
	r> @ nsprite ;

:+obj | 'ver 'usar nombre spr scale x y --
	'obj 'objetos p!+ >r r!+ r!+ r!+ r!+ r!+ r!+ r> ! ;

:inobj | x y -- o
	'objetos 0.05 p.in2d? ;

|--------- WALL
:wall
	>r
	r@+	r@+	fpos
	r@+ dup dim
	r@+ nsprite rdrop ;

:+wall | spr scale x y --
	'wall 'walls p!+ >r r!+ r!+ r!+ r> ! ;

|----------fxs
:r01 rand 0.01 mod ;

:fx
	>r
	80 dup dim
	r@+ dup r@+ -0.9 <? ( 3drop rdrop 0 ; )
	fpos

	'estre swap rnsprite
	r@+ r 12 - +! 	| sum the vel
	-0.0001 r +!	| gravity
	r@+ r 12 - +!
	rdrop
	;

:+fxs | x y --
	;

:+estrellas | x y --
	swap 'fx 'fxs p!+ >r
	r!+ r!+
	r01 r!+ r01 r!+
	rdrop
	;
|---------- principal
:fondo
	sw dup 2 >> + dup dim
	0 0 fpos
	pant nsprite
	;

:menuobjeto | adr --
	dup 'menuobj !
	dup 20 + @ 'mensaje strcpy
	1 'menuactivo !
	;

:moveto
	menuactivo 1? ( drop menuclick ; ) drop
	fxymouse
	2dup inobj 1? ( menuobjeto )( "" mensaje! ) drop
	'yd ! 'xd !
	xd xp - -? ( -200 )( 200 ) 'dz ! abs
	yd yp - -? ( 'seccaminaf )( 'seccaminad ) 'animap ! abs
	>? ( 'seccamina 'animap ! )
	drop ;


:resetpant
	'walls p.clear
	'objetos p.clear
	'fxs p.clear
	"" mensaje!
	;


:vermesa
	"La mesa parece bastante sucia!" mensaje! ;

:usamesa
	| agrega clip si no esta
	| agrega bollo de papel
	;

:pantalla1
	resetpant
|	'ob1 80 0.2 -0.5 +wall
|	0 0 "Boton" 'ob1 80 0.2 0.5 +obj
	0 'vermesa 
	"Mesa" 'mesa 300 -0.5 0.4 +obj
	'usamesa 'vermesa 
	"Mesa" 'mesa 300 -0.6 -0.1 +obj
	0 'vermesa 
	"Mesa" 'mesa 300 -0.5 -0.7 +obj
	;


:main
	pantalla1
	fonti2
	4
	show clrscr
|		fondo

		scr 'moveto guiBtn
		'salon1 nsprite

		'walls p.draw
		'objetos p.draw
		personaje
		'fxs p.draw

		drawmenu

		scr dup " %d " print
		'mensaje print

   		cflecha
		'exit >esc<
		;

:ram
	mark
	100 'objetos p.ini
	100 'walls p.ini
	2048 'fxs p.ini
	;

: ram main ;
