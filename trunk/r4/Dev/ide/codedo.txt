#sizepad

#error
#ypad

#tokenrun>
#pad )( 2048	| texto
#pad> 'pad

#hist )( $ffff	| historia
#hist> 'hist

:resetpad
	sh 2/ |sizepad -
	'ypad ! ;

:addhist
	'pad count swap hist> strcat 'hist> +! ;

|-----------------------------------------------

:showtokens
	'tokencod ( tokencod> <? )( @+
		|allowcr
		tokenstr printx sp
		) drop
	;

|-----------------------------------------------
:seterror
	;

#ee
:compila
	'pad tokenlast 0? ( drop ; )
	'ee !
	0 'pad !
	refreshfoco ;

:compila&run
	compila

	stackreset
	tokenrun> exectoken

	;

:showglass
	ypad sw * 2 << framev +
	sw sh ypad - *
	( 1? )( 1- swap dup @ 2/
		$3f3f3f and
		$f or
		swap !+ swap )
	2drop ;

:showpad
	dejavusans sizepad rfont!
	showglass
	4 ypad atxy | imprime tira de tokens
	cyan
	'tokencod ( tokencod> <? )( @+
		tokenstr allowcr printx sp
		) drop
	blanco
	 'pad 2048 inword
	'compila <spc>
	'compila&run <enter>
	| del
	| left
	| rigth
	[ 4 'sizepad +! resetpad ; ] <f1>
	[ -4 'sizepad +! resetpad ; ] <f2>
	;

:msgerror
	error 0? ( drop ; ) print ;

:showhistoria
	blanco
	'hist print
	;

#sizei 16

:btnico
	sizei 2* 2 - qdim drawric sizei 2* qdim ;

:showbotones
	sw sizei 2 + - ypad sizei + pos
	sizei 2* qdim
	negro 'exit btne rojo 'i.off btnico gc<<
	negro 'exit btne cyan 'i.file btnico gc<<
	negro 'exit btne verde 'i.bolt btnico gc<<
	negro 'exit btne amarillo 'i.time btnico gc<<
	negro 'exit btne violeta 'i.trash btnico
	;

:main
	4
	show clrscr
		fonti
		verde dup "r%d: " print cr
		rojo msgerror
		amarillo dumptok cr
		verde dumpvstack
		verde dumpdic cr

|		showhistoria
		showpad
		showbotones

		cminiflecha
		'exit >esc< ;

:m
	mark
	80 'sizepad !
	resetpad
	0 paper
	'tokencod 'tokenrun> !
	;

: m main ;
