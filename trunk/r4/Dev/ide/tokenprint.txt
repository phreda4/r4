|-- color de token

#codec $ffffff $ff0000 $ff0000 $ff00ff $ff00ff $7f7f7f $ffff00 $ffff00 $ffff00 $ffff00 $ffff00 $ffffff $ffffff $ff88ff $ffff $88ff

::tokencolor | a -- a
	dup $ff and
	16 <? ( 2 << 'codec + @ ink ; )
	drop verde ;

|-- token a string

:c1 tok>name ":%w" ;
:c2 tok>name "::%w" ;
:c3 tok>name "#%w" ;
:c4 tok>name "#:%w" ;
:c5 tok>mem "|%s" ;
:c6 tok>mem "^%w" ;
:c7 tok>cte "%d"  ;
:c8 tok>cte "$%h" ;
:c9 tok>cte "%%%b" ; | %% no permite mprint mprint
:cA tok>cte "%f" ;
:cB tok>mem """%s""" ;
:cC tok>name "%w" ;
:cE tok>name "'%w" ;

#codeis 0 c1 c2 c3 c4 c5 c6 c7 c8 c9 cA cB cC cC cE cE

::tokenstr | a -- mem
	dup $ff and 16 <? ( 2 << 'codeis + @ exec ; )
	nip 16 - macroname ;

|-- nombre a nro
:c1 8 >> ":w%h" ;
:c3 8 >> "#w%h" ;
:cC 8 >> "w%h" ;
:cE 8 >> "'w%h" ;
