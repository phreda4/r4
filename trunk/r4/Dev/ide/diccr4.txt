|---- diccionario
| string|token|mem|info+len+call+gr   16 bytes por palabra
| 0      4     8   12
#:indicepal )( $fffff
#:indicepal> 'indicepal
#:indicepal< 'indicepal

|---- memoria token
#:tokencod )( $ffff
#:tokencod> 'tokencod
#:tokencte )( $ffff
#:tokencte> 'tokencte
#:tokenmem )( $ffff
#:tokenmem> 'tokenmem


::dic>str	4 << 'indicepal + ;			| n -- %a
::dic>tok	4 << 'indicepal + 4+ ;		| n -- %a
::dic>mem	4 << 'indicepal + 8 + ; 	| n -- m
::dic>inf	4 << 'indicepal + 12 + ;	| n -- %a


::diccreset
	'indicepal dup 
	'indicepal> ! 'indicepal< !
	'tokencod 'tokencod> !
	'tokencte 'tokencte> !
	'tokenmem 'tokenmem> !
	;

::?palabra | str -- str dir / str 0
	0
	;


::?sistema | str -- str dir / str 0
	0
	;


::dumpdic
	'indicepal ( indicepal <? )(
		@+ allowcr print sp
		12 + ) drop
	;
