:setxymorton | morton --
	256 >r
	2 9 pick2 24 >> - 1- 2* << | mask	| 9 es quadsize
	xc yc rot 		| morton x y mask
	( 1? )( | morton x y mask
		pick3 and? ( swap r + swap ) 2/
		pick3 and? ( rot r + rot rot ) 2/
		r> 2/ >r )
	rdrop
	drop rot drop
	setxy ;

:fillquad | oct qua noct bit child 'oct -- oct qua noct bit
	nip
	octre - pixels + @  | getcolor
	pick3 | setqbuffer

	swap setxymorton | color size --
	dup ( 1? )( 1-
		over ( 1? )( 1-
			pick3 px!+
			) drop
		sw pick2 - px+!
		) 3drop ;

:fillpoint | oct qua --

	swap | setqbuffer
	setxymorton px!+ ;

:m3dto2d-child | oct qua noct bit child mask -- oct qua noct bit

:m3dto2d | oct qua --
	"> " ,s dup ,h ,sp
	( pntquad 1? )( 1- 				| oct qua mort

		$ff000000 nand? ( "p " ,s dup ,h ,cr fillpoint ; )	| si es punto..pinta
		) drop

	over @ dup | oct qua nodo nodo
	8 >> 2 << octre + swap $ff and 			|  oct qua noct bit
	dmask $7 and 1 over <<
	pick2 and? ( m3dto2d-child popchild )( 2drop )

:maskabs | ax ay az -- mask
	| ax ay az
	pick2 <? ( over <? ( | x<z && y<z
			drop <? ( 0 )( 8 )
			nip ; ) )
	rot | ay az ax
	pick2 <? ( over <? ( | y<x && z<x
			drop <? ( 16 )( 24 )
			nip ; ) )
	<? ( 32 )( 40 )
	nip nip ;

:calcmask

	0 0 1.0 transform
	rot +? ( 1 )( 0 ) >r abs
	rot +? ( 2 r+ ) abs
	rot +? ( 4 r+ ) abs
	maskabs r> or
	2 << 'maskc + @ 'dmask !
	;
