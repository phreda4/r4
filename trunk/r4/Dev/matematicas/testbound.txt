:nsqrt | v -- r
	0? ( ; )
	0 $40000000 | op res one
	( pick2 >? )( 2 >> )
	( 1? )(
		2dup +
		pick3 <=? (
			>r rot r> - | res one op
			rot pick2 2* +	| one op res
			2/ rot 2 >>
			)( drop 2 >> swap 2/ swap )
		)
	drop nip ;

:power2 | v -- v^2
	abs
	0 1 rot | t m a
	( 1? )(
		1 and? ( rot pick2 xor rot rot )
		swap 2 << swap 2/ )
	2drop ;

:main
	mark
	show clrscr verde
		1 ( 10 <? )(
|			dup sqrt over nsqrt pick2 "%d %d %d" print cr
			dup power2 over dup * pick2 "%d %d %d" print cr
			1+ ) drop
		'exit >esc< ;

: main ;
