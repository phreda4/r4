:drawimagen
	imagen >r
	res ( 1? )( 1-
		res ( 1? )( 1-
			r@+ px!+
			) drop
		sw res - px+!
		) drop
	rdrop
	;

:copyimg
	imagen >r
	res ( 1? )( 1-
		res ( 1? )( 1-
			px@ r!+ 1 px+!
			) drop
		sw res - px+!
		) drop
	rdrop
	;

|-- mouse
