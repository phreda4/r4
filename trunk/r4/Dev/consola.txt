^r4/lib/gui.txt

#wcon 32
#hcon 24

#consola )( $ffff

#wsum
#wcar
#hcar


:emitc | c --
	3 << dup 2/ + 'rom8x12 +
	12 ( 1? )( 1-
		swap c@+
		$80 and? ( ink@ px!+ )( 1 px+! )
		$40 and? ( ink@ px!+ )( 1 px+! )
		$20 and? ( ink@ px!+ )( 1 px+! )
		$10 and? ( ink@ px!+ )( 1 px+! )
		$8 and? ( ink@ px!+ )( 1 px+! )
		$4 and? ( ink@ px!+ )( 1 px+! )
		$2 and? ( ink@ px!+ )( 1 px+! )
		$1 and? ( ink@ px!+ )( 1 px+! )
		drop
		wcar px+!
		swap )
	2drop ;

:nemitc | c --
	3 << dup 2/ + 'rom8x12 +
	12 ( 1? )( 1-
		swap c@+
		$80 nand? ( ink@ px!+ )( 1 px+! )
		$40 nand? ( ink@ px!+ )( 1 px+! )
		$20 nand? ( ink@ px!+ )( 1 px+! )
		$10 nand? ( ink@ px!+ )( 1 px+! )
		$8 nand? ( ink@ px!+ )( 1 px+! )
		$4 nand? ( ink@ px!+ )( 1 px+! )
		$2 nand? ( ink@ px!+ )( 1 px+! )
		$1 nand? ( ink@ px!+ )( 1 px+! )
		drop
		wcar px+!
		swap )
	2drop ;

:drawcon
	'consola >r
	0 0 setxy
	hcon ( 1? )( 1-
    	wcon ( 1? )( 1-
    		r@+ dup 8 >> ink
    		$100 and? ( $ff and emitc )( $ff and nemitc )
			hcar px+!
			) drop
    	wsum px+!
    	) drop
	rdrop
	;

:randcon
	'consola >r
	wcon hcon *
	( 1? )( 1- random r!+ ) drop
	rdrop ;

:fullcon
	sw 3 >>  'wcon !
	sh 12 /  'hcon !
	sw 12 * wcon 8 * - 'wsum !
	sw 8 - 'wcar !
	sw 12 * neg 8 + 'hcar !
	;

:main
	fullcon
	randcon
	show clrscr
		drawcon
		'exit >esc<
		'randcon <f1>
		cminiflecha
	;

: main ;
