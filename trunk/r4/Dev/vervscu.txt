| verduras vs cucarachas
^r4/lib/gui.txt
^r4/lib/trace.txt
^r4/lib/part16.txt
^r4/dev/grafico1.spr

#seccuca cuca1 cuca2
#seczana zanaoria1 zanaoria2 zanaoria3 zanaoria4 zanaoria5

#cucarachas 0 0
#verduras 0 0
#disparos 0 0

:memoria
	mark
	100 'cucarachas p.ini
	100 'verduras p.ini
	100 'disparos p.ini
	;


:haceverduras 	'verduras p.draw ;
:hacecucarachas 'cucarachas p.draw ;
:hacedisparos	'disparos p.draw ;

:cucar | a --
	>r
	r@+ r@+
	r@+ r 12 - +!
	r@+ r 12 - +!
	rdrop
	0.6 >? ( pick2 'cucarachas p.del )
	fpos
	200 dup dim
	msec 5 >> $4 and 'seccuca + @ nsprite
	;

:dispa | a --
	>r
	r@+ r@+
	over sw 2/ - abs
	over sh 2/ - abs +
	20000 >? ( drop 0 rot rot )( drop )
	fpos
	200 dup dim
	'espora r 4 - @ rnsprite
	r@+ r 12 - +!
	r@+ r 12 - +!
	rdrop
	;

:verdu | a --
	>r
	200 dup dim
	r@+ r@+ fpos
	msec 5 >> $c and 'seczana + @ nsprite
	r@+ r 12 - +!
	r@+ r 12 - +!
	rdrop
	;

:r.01 rand 0.01 mod ;

:+cuca
	'cucar 'cucarachas p!+ >r
	r.01 r.01 0 0 r!+ r!+ r!+ r!+ rdrop ;

:+disp
	'dispa 'disparos p!+ >r 
	r.01 r.01 0 0 r!+ r!+ r!+ r!+ rdrop ;
	;

:+verd
	'verdu 'verduras p!+ >r 
	r.01 r.01 0 0 r!+ r!+ r!+ r!+ rdrop ;

:inicio
	33
	show clrscr
		dup "%d" print
		haceverduras
		hacecucarachas
		hacedisparos
		'exit >esc<
		'+cuca <f1>
		'+disp <f2>
		'+verd <f3>
		;

: memoria inicio ;
