^r4/dev/simple.spr

#bubles 0 0

:resetgame
	'bubles p.clear
	;

|---------
:r.01 rand 0.001 mod ;
:r.1 rand 0.1 mod ;
:r.8 rand 0.8 mod ;

:bub | adr -- adr/adr 0 delete
 	>r
	r@+ dup abs
	0.99 >? ( r 4+ dup @ neg swap ! ) drop
	r@+
	-0.9 <? ( -0.9 r 4 - ! r 4+ dup @ neg swap ! ) 
	fpos
	r@+ r 12 - +!
	r @ 0.001 - r ! | gravedad
	r@+ r 12 - +!
	rdrop
	0.08 dup fdim
	'coso1 nsprite
	;

:+buble
	'bub 'bubles p!+ >r
	r.8 r.8 r!+ r!+
	r.01 r.01 r!+ r> ! ;

:collision | p1 p2 -- p1 p2
  	over 4+ @
	over 4+ @ - dup *.
	pick2 8 + @
	pick2 8 + @ - dup *. +
	0.02 >? ( drop ; )
	"%f " print ;

:collision | p1 p2 -- p1 p2
	over 4+ @ over 4+ @ - dup *. | (x1-x2)^2
	pick2 8 + @ pick2 8 + @ - dup *. +
	0.2 >=? ( drop ; ) sqrt. 0.1 swap -
	2/ >r
	over 4+ @ over 4+ @ -
	pick2 8 + @ pick2 8 + @ -
	atan2 sincos swap				| p1 p2 si co
	dup r *. pick4 4+ +!
	dup r *. neg pick3 4+ +!
	over r *. pick4 8 + +!
	over r *. neg pick3 8 + +!

	dup r *. pick4 12 + +!
	dup r *. neg pick3 12 + +!
	over r *. pick4 16 + +!
	over r *. neg pick3 16 + +!

	2drop
	rdrop
	;

:main
	resetgame
	show clrscr
		scr home blanco
		dup "%d " print
		'bubles p.draw
		'collision 'bubles p.map2
		'+buble <spc>
		'exit >esc< ;

:inicio
	mark
	1000 'bubles p.ini
	;

: inicio 33 main ;
