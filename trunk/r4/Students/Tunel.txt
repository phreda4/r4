:posHombre

	xh 2 - 0.2 * 0.5 -
	yh 2 - 0.2 * 0.5 -

:dibujaH

	vxH 0.203 fdim
	posHombre
	s1 nsprite

:cambia

	yh xh xymapa c! 
;

:abrepuerta
 
	puntaje 1 + 'puntaje !
	conta 1 + 'conta !
	conta 2 =? ( 5 xmapa ymapa xymapa c! )
	drop

;

:choca

	xh sentido + 'xh !
;

:chocay

	yh sentido + 'yh !
;

:desplazabloque

	yh 1 + 'ym !
	xh sentido + 'xm !   
	xh 0 >=? ( 
				9 <=? ( 
						drop yh 9 <? ( 
										drop ym xm xymapa c@ 1 =? (
																	drop yh xm xymapa c@ 0 =? (
																								drop 0 ym xm xymapa c! 
																								1 yh xm xymapa c! ; ) ) ) ) )
	drop 
;

:desplazabloquey
	
	yh + 1 'ym !
	yh 8 <? ( 
				drop ym 1 + xh  xymapa c@ 1 =? ( 
												drop ym xh xymapa c@ 0 =? ( 
																			drop 0  ym 1 + xh  xymapa c! 1 ym xh  xymapa c! ; ) ) )
   drop 
;

:y-

	yh 1 -
;

	yhnivel 'yh ! 
	contanivel 'conta ! 
	drop 
;

:ascensor
	nivel 4 =? (
				( yh 4 <? )(
							1 + 'yh !   | yh+ yh+ -- yh+
							 20 yh 1 - xh xymapa c!
							 19 yh xh xymapa c!
							)  
				drop
				)
	drop                        
;

:tomapico

	pico 0 =? (
				28 yh xh xymapa c!
				yh 'dirp1 !
				xh 'dirp2 !
				'dib1C 's1 !
				1 'pico ! 
			  )
	drop
;

:picar

	'dib1C 's1 !
	pico 1 =? (
				0 dirp1 dirp2 xymapa c!
				0 yh xh xymapa c!
				0  'pico ! 
			   )
	drop
;

:opcionesx

	 3 =?  ( 0 cambia ; )( 
		 1 =?  ( choca ; )(
			 12 =? ( choca ; )(
				 4 =?  ( choca ; )( 
					 5 =?  ( pasanivel ; )(
						 9 =?  ( 0 cambia abrepuerta ; )(
							 10 =? ( 0 cambia abrepuerta ; )(
								 11 =? ( 0 cambia abrepuerta ; )(
									 19 =? ( ascensor ; )(
										 25 =? ( tomapico ; )(
											 29 =? ( picar ; )(
												 2 =?  ( xh 'xh ! ; ) ) ) ) ) ) ) ) ) ) ) )   
;


:opcionesy

	3 =?  ( 0 cambia ; )(
		1 =?  ( chocay ; )(
			12 =? ( chocay ; )(
				4 =?  ( chocay ; )(
					5 =?  ( pasanivel ; )(
						9 =?  ( 0 cambia abrepuerta ; )(
							10 =? ( 0 cambia abrepuerta ; )(
								11 =? ( 0 cambia abrepuerta ; )(
									19 =? ( ascensor ; )(
										25 =? ( tomapico ; )(
											29 =? ( picar ; )(
												2 =?  ( xh 'xh ! ; ) ) ) ) ) ) ) ) ) ) ) )
;

:cae

	xh 0 >=? ( 9 <=? ( drop
						yh xh xymapa c@  2 <>? (  							| valor 2 -- valor
												 yh 1 - dup xh xymapa c@   	| y- y- x -- y- valor
												 0 =? ( 
														drop 'yh ! drop ;
													)
												2drop ) ) )   				| y- valor  --
drop
;

:caminader

	  0.203 'vxH !
	  yh xh xymapa c@ | valormapa
	  opcionesx
	  drop
;

:caminaizq

	-0.203 'vxH !
	yh xh xymapa c@ 	| valormapa
	opcionesx
	drop
;

:subeh

  yh xh xymapa c@ 0 <>? ( 25 <>? (
								   yh 1 + dup 'yh ! xh xymapa c@ | valormapa
								   opcionesy
								   drop ) )
  drop
;

:bajah

	yh -? ( 0 'yh ! )
	drop
	yh xh xymapa c@	 | valormapa
	opcionesy
	drop
;
