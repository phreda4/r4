:vala
	actv @+ 0? ( 2drop 'act @+ )
	swap 'actv ! ;

:initr
	dup >r
	0 0 0 0 0 0 'new3 |vala
	1.0 r!+ r!+ r!+ r!+ r!+ r!+ r!+ r> !
|	rand 8 >> $ff and 1+ ( 1? )( 1- over rstep ) drop 
	;

	mpush
	over $7 and 3 - 0.9 *
	pick2 3 >> $7 and 3 - 0.9 *
|	2dup "%d %d" print cr
	0 mtransi
	dup rplay
	mpop ;

		'freeze <f2>
