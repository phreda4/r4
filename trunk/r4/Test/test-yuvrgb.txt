^r4/lib/gui.txt
^r4/lib/color.txt

#yy
#hh
#sx #sy


:drawyuv
	framev >r
	0 ( 256 <? )(
		0 ( 256 <? )(
			yy pick2 pick2 yuv32 r!+
			1+ ) drop
		sw 256 - 2 << r+
		1+ ) drop
	rdrop
	;

:drawy
	drawyuv
	;

#yyy
:slidey
	'yyy hslide
	yyy 0.5 + | 0..1.0
	256 16 *>> 'yy !
	;

:drawhsv
	1.0 w / 'sx !
	1.0 h / 'sy !
	xc w 2/ - yc h 2/ -
	0 ( 1.0 <? )(
		pick2 pick2 setxy
		0 ( 1.0 <? )(
			hh pick2 pick2 hsv2rgb px!+
			sx + ) drop
		swap 1+ swap
		sy + ) drop
	2drop
	;

:drawh
	drawhsv
	;

#hhh
:slideh
	'hhh hslide
	hhh 0.5 + 'hh !
	;

:inicio
	inigui
	'exit >esc<
	show clrscr
		drawyuv
		scr
0.5 0.5 fdim -0.5 -0.5 fpos drawy
0.5 0.1 fdim -0.5 0.8 fpos slidey
0.4 0.4 fdim 0.5 -0.5 fpos drawh
0.5 0.1 fdim 0.5 0.8 fpos slideh

		fonti home blanco
		dup "%d" print cr
		yy yyy "Y:%f %d" print cr

		cmano
		;

: 0 paper inicio ;
