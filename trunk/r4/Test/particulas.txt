^inc/escena.inc

|------------------------------------------
|  Ax    Vx   Px    Ay   Vy   Py  D   A  Ang
#particulas  )( 32768
#particulas> 'particulas

|-------------------------
:borralinea | dir -- dir++
	28 - dup | dirc dirc --> tengo la direccion de comienzo dos veces
	particulas> 36 - 36    | dirc dirc fun cant
	move  | dirc
	-36 'particulas> +!
	;

|------------------------------
:calcAVP  | 'dir -- pos 'dir+
	@+ over +!
	@+ over +!
	@+ swap ;

|--------------------------------------
:dibpart  | spr dir(tabla) -- dir+(siguiente)
	calcAVP calcAVP >r fpos
	r@+ dup fdim
	r @ 2 - -? ( drop r> borralinea ; )
	dup r!+ alpha		| desvaneciendose
	dup
	r @ 0.08 + dup r!+	| rotando
	rnsprite r>
	;

|------------------------------------------
:dibujarparticulas | spr --
	'particulas
	( particulas> <? )(	dibpart ) drop
	$ff alpha drop ;

|------------------------------------------
:+particulas
	particulas> !+ !+ !+ !+ !+ !+ !+ $ff swap !+ 0 swap !+ 'particulas> !
	;

|-------------------------------------------------
:gv
	rand 3 >> 0.06 mod ;

:ga
	dup 3 >> neg ;

:dimr
	rand 3 >> 0.15 mod ;

#fps 0
#rfps
:main
	inigui
	'exit >esc<
	showo clrscr
		'dib13 dibujarparticulas
|		40 .mseg .restart
		scr
		dimr xymousef swap gv ga rot gv ga +particulas
		dimr xymousef swap 0 0 rot 0 0 +particulas
|---------- debug + cursor
		blanco scr fonti home
		dup "%d" print cr
		rfps "%d" print cr
		particulas> 'particulas - 36 / "%d" print
		cflecha
		1 'fps +!
		1 .segs .restart
		fps 'rfps ! 0 'fps !
 ;

: 33 main ;
