^libgui.txt
^libsprite.txt

#cx #cy
#gy #gx

:ramp
	-? ( [ neg ; ] >r neg )
	$7fff >? ( $7fff nip )
	$7fff swap 0? ( 1 nip ) / ;

|-----------------------------------
:dibujadeg
	sw 2/ sh 2/ fcen  | centro degrade
	xymouse
	sw 2/ rot -
	sh 2/ rot -

	2dup 2dup distfast 0? ( 1 nip ) >r atan2 $3fff + sincos r / swap r> / |...1era version

	'gy ! 'gx !

|	2drop 0 0 |...null

|	swap neg

|	neg swap neg swap

|	sh 4 *>> swap neg sw 4 *>>  swap

	ramp swap neg ramp swap

	2dup 'cy ! 'cx ! | registro


	fmat
	$ff0000 $00ff00 fcol
	lfill
	gc.fbox
	sfill
	azul
	xymouse op
	sw 2/ sh 2/ line
	;

:calcn
	2dup distfast 0? ( 1 nip )
	>r atan2 $3fff + sincos r / swap r> /
	;



|-----------------------------------
:botoncontrol
	verde fonti
	blanco dup "%d" cr2 printc ;

:control3
	fonti blanco
	xymouse 2dup "%d %d " print cr
	sw 2/ rot -
	sh 2/ rot - "%d %d" print cr
	cy cx "x:%d y:%d" print cr
	gy gx "g:%d g:%d" print cr

	|5 'botoncontrol .hlist
	;

|-----------------------------------
:botonsalir
	'exit "Salir" $ff0000 .btnl ;

|----------  Main
#guimenu
0.8 0.8 0.0 0.0 'dibujadeg
0.9 0.1 0 0.45 'control3
0.1 0.1 0.45 -0.45 'botonsalir
0

:main
	inikey
	'exit >esc<
	show cls scr
		'guimenu gui
		cmano
		;
