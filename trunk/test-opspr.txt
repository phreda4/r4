^libgui.txt
^libsprite.txt

#cx #cy
|-----------------------------------
:dibujadeg
	sw 2/ sh 2/ fcen  | centro degrade
	xymouse
	sw 2/ rot -
	sh 2/ rot -
	2dup distfast 0? ( 1 nip )
	>r 
	atan2 $3fff +

	sincos r / swap r> /
	2dup 'cy ! 'cx !

	2drop 0 0 fmat
	$ff0000 $00ff00 fcol
	lfill
	gc.fbox
	sfill
	;

:calcn
	2dup distfast 0? ( 1 nip )
	>r atan2 $3fff + sincos r / swap r> /
	;


|-----------------------------------
:botoncontrol
	verde fonti
	blanco dup "%d" cr2 printc ;

:control3
	fonti blanco
	xymouse 2dup "%d %d " print cr
	sw 2/ rot -
	sh 2/ rot - "%d %d" print cr
	cy cx "x:%d y:%d" print cr
	|5 'botoncontrol .hlist
	;

|-----------------------------------
:botonsalir
	'exit "Salir" $ff0000 .btnl ;

|----------  Main
#guimenu
0.8 0.8 0.0 0.0 'dibujadeg
0.9 0.1 0 0.45 'control3
0.1 0.1 0.45 -0.45 'botonsalir
0

:main
	inikey
	'exit >esc<
	show cls scr
		'guimenu gui
		cmano
		;


|****** BOOT ******
:	0 paper
	4 main
	"main.txt" run
	;
