#:foconow

::show 
	msec dup 'lastime !+ ! 0 '.exit !
	-1 'foconow !
	( .exit 0? )( drop
		update 
		r exec
		lastgui 1? ( redraw )( 1 'lastgui ! -1 'foconow ! ) drop )
	drop rdrop
	0 'lastgui ! 0 '.exit ! 0 key! ;

	show clrscr
		$fff ink
		10 10 op
		200 100 line
		;

: main ;
