| pruebas de colores
| PHREDA
|----------------------------------------
^reda4.txt
^libgui.txt
^libcolor.txt

#s 0
#l 255
#rgb
#cor
#cg
#cb
#yy

:drawyuv
	framev >r
	0 ( 256 <? )(
		0 ( 256 <? )(
			yy pick2 pick2 yuv32 r!+
			1+ ) drop
		sw 256 - 2 << r+
		1+ ) drop
	;

:drawhsv
	0 ( 1.0 <? )(	
		setxy
		0 ( 1.0 <? )(
			hh pick2 pick2 hsv2rgb px!+
			0.01 + ) drop
		0.01 + ) drop
	;

| colores con menos bits
:full24
	0 ( 256 <? )(
		dup 8 << ink 0 over op 100 over line	| 24bit
		dup $fe and 8 << ink 200 over line	| 21
		dup $fc and 8 << ink 300 over line	| 18
		dup $f8 and 8 << ink 400 over line	| 15
		1+ )
	;

#yyy
:slidey
	'yyy .hslide
	yyy 0.5 + | 0..1.0
	256 16 *>> 'yy !
	;

#guiyuv
0.8 0.1 0.0 0.40 'slidey
0

:inicio
	inikey scr
	'exit >esc<
	show cls
		drawyuv
		scr
		'guiyuv gui
		fonti blanco
		dup "%d" print cr
		yy yyy "Y:%f %d" print cr

		cmano
		;

: 0 paper inicio "main.txt" run
