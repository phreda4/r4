| pruebas de colores
| PHREDA
|----------------------------------------
^reda4.txt
^libgui.txt
^libcolor.txt

#s 0
#l 255
#rgb
#cor
#cg
#cb
#yy

:drawyuv
	framev >r
	0 ( 256 <? )(
		0 ( 256 <? )(
			yy pick2 pick2 yuv32 r!+
			1+ ) drop
		sw 256 - 2 << r+
		1+ ) drop
	;

| colores con menos bits
:full24
	0 ( 256 <? )(
		dup 8 << ink 0 over op 100 over line	| 24bit
		dup $fe and 8 << ink 200 over line	| 21
		dup $fc and 8 << ink 300 over line	| 18
		dup $f8 and 8 << ink 400 over line	| 15
		1+ )
	;

:inicio
	inikey scr
	'exit >esc<
|	3 16 .table 1 11 .at
|	blanco
|	255 's .hsli .\
|	1 2 .at
|	rojo 	255 'cor .hsli .\
|	verde	255 'cg .hsli .\
|	azul 	255 'cb .hsli .\
	show cls
|		s 'yy !
		drawyuv
|		full24
		fonti blanco
		dup "%d" print cr
|		h l s  "s:%d l:%d h:%h" print cr
|		cor "%h" rojo print
|		cg "%h" verde print
|		cb "%h" azul print
		;

: 0 paper inicio "main.txt" run
