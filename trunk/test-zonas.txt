^reda4.txt
^libgui.txt
^font.txt
^libsprite.txt

|-----------------------------------
:accionfile
	azul 2 wtext
	blanco 2dup "%d,%d" cr2 printc ;

:led
	rand ink gc.fbox
	;

:filelist
	64 64 'led .table
|	3 9 'accionfile .table
	;

|-----------------------------------
:botoncontrol
	verde 2 font
	blanco dup "%d" cr2 printc ;

:control3
	5 'botoncontrol .hlist ;

|-----------------------------------
:botonsalir
	'exit "Salir" $ff0000 .btnl ;

|----------  Main
#guimenu
0.8 0.8 0.0 0.0 'filelist
0.9 0.1 0 0.45 'control3
0.1 0.1 0.45 -0.45 'botonsalir
0

:main
	inikey
	'exit >esc<
	show cls scr
		'guimenu gui
		scr 8 font
		dup ":r%d" verde print
		cmano
		;


|****** BOOT ******
:	0 paper
	4 main
	"main.txt" run
	;
