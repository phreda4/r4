^reda4.txt
^gui3.txt

|-----------------------------------
:accionfile
	azul 2 wtext
	blanco 2dup "%d,%d" cr2 printc ;

:led
	rand ink gc.fbox
	;

:filelist
	64 64 'led table
|	3 9 'accionfile table
	;

|-----------------------------------
:botoncontrol
	verde 2 font
	blanco dup "%d" cr2 printc ;

:control3
	5 'botoncontrol hlist ;

|-----------------------------------
:botonsalir
	rojo 'exit 'ifin .btnd ;

|----------  Main
#guimenu
0

:main
	inigui
	'exit >esc<
	show clrscr
 	     0.8 0.8 fdim 0.0 0.0 fpos filelist
     	     0.9 0.1 fdim 0 0.9 fpos control3
	     0.1 0.1 fdim 0.9 -0.9 fpos botonsalir
		scr 8 font
		dup ":r%d" verde print
		cmano
		;


|****** BOOT ******
:	0 paper
	4 main
	"main.txt" run
	;
