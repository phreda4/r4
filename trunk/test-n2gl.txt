#xcam 0.0
#ycam 0.0
#zcam 10.0

#fps 0
#rfps

:main
	0 paper
	inigui

	'exit <esc>
	showo clrscr
		1.0 3dmode
		xcam ycam zcam mtransi
		papared techo pipiso
		mpush
		x y z 10.0 +  mtrans
		'dib7 3dnsprite
		mpop

		x xv + dup abs 12.0 <? ( drop 'x ! )( 2drop xv neg 'xv ! )
		y yv + dup abs 12.0 <? ( drop 'y ! )( 2drop yv neg 'yv ! )
		z zv + dup abs 12.0 <? ( drop 'z ! )( 2drop zv neg 'zv ! )
		xv 2/ neg 'xcam +!
		yv 2/ neg 'ycam +!
|		newcubo
		4 'xm +!

		fonti
		"close2gl" verde print "demo" rojo print
		blanco cr
		z y x "x:%f y:%f z:%f" print cr
		dup "%d" print cr
		rfps  "fps:%d" print cr

		0.1 1.0 3.1415 0.002 0.02 0.2 "%f %f %f %f %f %f" print cr

		1 'fps +!
		1 .segs .restart
		fps 'rfps ! 0 'fps !

		;

: 33 main "main.txt" run ;
