#bitmaptecho )( $1000

:techo
	mpush
|	$1ff mrotx
	-10.0 -5.0 20.0  mtrans
	20 ( 1? )( 1-
		20 ( 1? )( 1-
|			$1ff mrotxi
|			2dup 10 * + bitmaptecho + c@  mrotxi
			2dup 10 * + bitmaptecho + c@ dup
			0 swap  13 << 0 mtrans
			'dib5 3dnsprite
			0 swap neg 13 << 0 mtrans
			1.0 0.0 0.0 mtrans
			) drop
		-20.0 0.0 -1.0 mtrans
		) drop
